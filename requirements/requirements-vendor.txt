# Third-party runtime dependencies vendored into the repo-root vendor/ and added to
# sys.path at add-on startup. Anki does NOT pip-install for you, so anything here is
# bundled with the add-on.
#
# RULE: everything here is pure-Python and cross-platform — no compiled wheels (they break
# on the other OS). The provider HTTP layer is stdlib-only; these deps exist for the
# typed-config layer and the gemini_vertex auth path.

# --- Typed config (Pydantic v1 + TOML) ------------------------------------------------
# Pydantic v1, not v2: v2 depends on the compiled (Rust) `pydantic_core` wheel, which is
# not pure-Python and cannot ship in one cross-platform .ankiaddon. v1 has a pure-Python
# core, so `scripts/vendor_deps.py` installs it with `--no-binary pydantic` to force the
# pure-Python build.
pydantic>=1.10,<2     # typed/validated config models (pure-Python)
tomli-w>=1.0          # write config TOML (reading uses stdlib tomllib on 3.11+)

# --- gemini_vertex service-account auth (RS256 JWT) -----------------------------------
rsa>=4.9               # RS256 signing of the JWT assertion (pure-Python)
pyasn1>=0.6            # unwrap the PKCS#8 service-account private key to PKCS#1 (pure-Python)
# (Only needed for gemini_vertex with credentials_path/credentials_json; the access_token
#  strategy needs nothing vendored.)
#
# After editing this file run:  python scripts/vendor_deps.py
